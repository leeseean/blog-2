### 12.单词首字母大写
```js
String.prototype.toJadenCase = function () {
  return this.replace(/(^|\s)[a-z]/g, function(s){ return s.toUpperCase(); });
};

String.prototype.toJadenCase = function () { 
  return this.split(" ").map(function(word){
    return word.charAt(0).toUpperCase() + word.slice(1);
  }).join(" ");
};
```
