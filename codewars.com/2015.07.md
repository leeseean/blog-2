# pattern
pattern(8)输出
```html
22
4444
666666
88888888
```
```js
function pattern(n) {
    var res = [], i = 1;
    while (i++ < n) {
        res.push(Array(i + 1).join(i++)); // i+1 array的join会少一位
    }
    return res.join('\n');
}
```

```js
function pattern(n) {
 var number = "", output = "";
 while (n > 0) {
   number = n + number;
   output = number + (output ? "\n" : "") + output;
   n--;
 }
 return output;
}
```

# reverseIt
`"Hello" -> "olleH"`
```js
function reverseIt(data){
  if(typeof data === 'string'){
    return data.split("").reverse().join("");
  }
  if(typeof data === 'number'){
    var a = +(data.toString().split("").reverse().join(""));
    return a;
  }
  return data;
}
```
